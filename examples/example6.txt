int main ( ) { 
int N = 1024 ; 
int buf [ N ] ; 
int acc = 0 ; 
int accu = 0 ; 
 
for ( int i = 0 ; i < N ; i ++ ) { 
buf [ i ] = i ; 
}
 
for ( int i = 0 ; i < N ; i += 1 ) { 
acc += buf [ i ] ; 
} 
acc *= 1 ;  
 
for ( int i = 0 ; i < N ; i += 4 ) { 
accu += buf [ i ] ; 
} 
accu *= 4 ;  
 
return 0 ; 
}
